```css
  --badge-amrap-border: rgba(168, 85, 247, 0.3);
  /* ... other badge tokens ... */
}

.workout-type-badge.amrap {
  background-color: var(--badge-amrap-bg);
  color: var(--badge-amrap-text);
  border: 1px solid var(--badge-amrap-border);
}
```
