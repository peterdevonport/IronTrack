```bash
MERGEABLE=$(echo "$decoded" | jq -r '.mergeable')
CONCLUSION=$(echo "$decoded" | jq -r '.conclusion')
STATUS=$(echo "$decoded" | jq -r '.status')
```
Or even better, extract all fields in one jq invocation:
```bash
eval "$(echo "$row" | base64 --decode | jq -r '@sh "PR_NUM=\(.number) BRANCH=\(.branch) MERGEABLE=\(.mergeable) CONCLUSION=\(.conclusion) STATUS=\(.status)"')"
```
