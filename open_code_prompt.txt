### Location
Found in recent master commits. File: `app.js`, lines 83-113 (the `window.addEventListener('popstate', ...)` handler).

### Description
The popstate handler contains 4 levels of nesting with 4 different return paths:

```javascript
window.addEventListener('popstate', (e) => {
  if (appView.classList.contains('hidden')) {     // Path 1
    if (history.length > 1) { history.back(); }
    return;
  }
  const s = e.state;
  if (profileModal && !profileModal.classList.contains('hidden')) {  // Path 2
    closeProfileModal(true);
    if (s?.tab) { activateTab(s.tab); }
    return;
  }
  if (s?.modal === 'profile' && profileModal?.classList.contains('hidden')) {  // Path 3
    openProfileModal();
    return;
  }
  if (!s) return;   // Path 4
  if (s.tab && s.tab !== state.ui.currentTab) {   // Path 5 (no return)
    activateTab(s.tab);
  }
});
```

This control flow is difficult to reason about because:
1. The conditions are not mutually exclusive in obvious ways (e.g., both modal checks reference `profileModal`).
2. The `history.length > 1` guard uses a magic number without documentation.
3. There's a subtle interaction: when the modal is closing via popstate, it calls `closeProfileModal(true)` which skips `history.back()`, but the modal state restoration depends on `e.state` which is the state of the page being navigated TO, not FROM. The correctness relies on understanding browser history semantics that aren't documented here.
4. Missing edge case: if `e.state` is null (e.g., fresh navigation to the page), the function silently does nothing — no tab activation, no UI update.

### Suggested Refactor
Extract each return path into a named function to flatten the nesting and make the intent explicit:

```javascript
function handlePopstateAppHidden() {
  if (history.length > 1) history.back();
}

function handlePopstateModalClose(state) {
  closeProfileModal(true);
  if (state?.tab) activateTab(state.tab);
}

function handlePopstateModalRestore() {
  openProfileModal();
}

function handlePopstateTabSwitch(state) {
  if (state?.tab && state.tab !== state.ui.currentTab) {
    activateTab(state.tab);
  }
}

window.addEventListener('popstate', (e) => {
  if (appView.classList.contains('hidden')) return handlePopstateAppHidden();
  const s = e.state;
  if (isModalOpen()) return handlePopstateModalClose(s);
  if (isModalRestore(s)) return handlePopstateModalRestore();
  handlePopstateTabSwitch(s);
});
```
